#include <bits/stdc++.h>
using namespace std;
int main() {
    int n;
    long long d,t;
    cin>>n>>d>>t;

    vector<pair<long,long>>v(n);
    for (int i=0;i<n;i++) {
        cin>>v[i].first>>v[i].second;
    }
    
    sort(v.begin(), v.end(), [](pair<long,long>&a , pair<long,long>&b){
        if(a.first!=b.first){
            return a.first<b.first;
        } 
        else{
            return a.second<b.second;
        }
    });
    
    vector<long long> a(n);
    for (int i=0; i<n; i++){
        a[i]=v[i].first;
    } 
    
    long long s=0;
    for (int i=0;i<n;i++) {
        long long q=v[i].first;
        long long p=v[i].second;
        auto it = upper_bound(a.begin(), a.end(), q+d);
        long long T;
        if (it==a.end()){
            T=t;
        } 
        else{
            T= *it;
        }

        if(q+p<=T){
            s++;
        } 
    }
    cout<<s<<endl;


}
